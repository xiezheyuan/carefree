149 template <class T1, class T2>
150 void raise(carefree_exception<T1, T2> err) {
151 switch (get_exception_policy()) {
152 case Throw:
153 throw err.get_err();
154 break;
155 case Ignore:
156 std::fprintf(stderr, "CareFree Library : An Error Occured.\n%s\n", err.what().c_str());
157 break;
158 case Friendly:
159 std::fprintf(stderr, "CareFree Library : An Error Occured.\n%s\n", err.what().c_str());

161 break;
162 case Simulate:
163 std::fprintf(stderr, "CareFree Library : An Error Occured.\n%s\n", err.what().c_str());
164 std::terminate();
165 break;
166 default:
167 __builtin_unreachable();
168 std::abort();
169 break;
170 }

364 err_positive_checker(length, __func__, "length");
365 std::vector<T> data;

367 int tmp = uniform(0.0, total / i * 2);

632 bool is_tree(graph g) {
633 std::vector<int> fa(g.N + 1);
634 for (int i = 1; i <= g.N; i++) fa[i] = i;
635 auto edges = g.get_edges();
636 if (edges.size() != (unsigned)(g.N - 1)) return false;
637 std::function<int(int)> find = [&](int x) {
638 return fa[x] == x ? x : fa[x] = find(fa[x]);
639 };
640 for (auto i : edges) {
641 int u = i.from, v = i.to;
642 if (find(u) == find(v)) return false;
643 fa[find(u)] = find(v);
644 }
645 return true;
646 }

659 graph prufer_decode(int n, std::vector<int> prufer, _Weight weightL = 0, _Weight weightR = 0) {
660 std::vector<int> deg(n + 1, 1);
661 prufer.insert(prufer.begin(), 0);
662 graph g(n, false);
663 if (prufer.size() != (unsigned)(n - 1)) raise(carefree_invalid_argument("prufer_decode : prufer size must be n-1"));
664 for (int i = 1; i <= (n - 2); i++) {
665 err_inrange_checker(prufer[i], 1, n, __func__, "prufer[" + std::to_string(i) + "]");
666 }
667 for (int i = 1; i <= (n - 2); i++) deg[prufer[i]]++;
668 int ptr = 1, leaf = 0;
669 while (deg[ptr] != 1) ptr++;
670 leaf = ptr;
671 for (int i = 1; i <= (n - 2); i++) {
672 int v = prufer[i];
673 g.add(v, leaf, randint(weightL, weightR));
674 if (--deg[v] == 1 && v < ptr)
675 leaf = v;
676 else {
677 ptr++;
678 while (deg[ptr] != 1) ptr++;
679 leaf = ptr;
680 }
681 }
682 g.add(n, leaf, randint(weightL, weightR));
683 return g;
684 }

688 std::vector<int> depth(g.N + 1);
689 q.push(1);
690 depth[1] = 1;

692 auto u = q.front();
693 q.pop();
694 for (auto i : g.get_edges(u)) {
695 if (!depth[i.to]) {
696 depth[i.to] = depth[u] + 1;
697 q.push(i.to);
698 }
699 }
700 }
701 return depth;
702 }

1697 judge_result limited_run(limprog prog, const string& input_file, const string& output_file) {
1698 auto start = std::chrono::system_clock::now();
1699 ull memory = 0, tim = 0;
1700 process proc(prog.program, input_file, output_file);
1701 while (proc.is_running()) {
1702 auto now = std::chrono::system_clock::now();
1703 tim = std::chrono::duration_cast<std::chrono::nanoseconds>(now - start).count();
1704 if (tim > prog.time) {
1705 proc.kill();
1706 return {TimeLimitExceeded, "time limit exceeded.", 0.0, tim, memory};
1707 }
1708 memory = std::max(memory, proc.memory());
1709 if (memory > prog.memory) {
1710 proc.kill();
1711 return {MemoryLimitExceeded, "memory limit exceeded.", 0.0, tim, memory};
1712 }
1713 }
1714 if (proc.return_code() != 0) {
1715 return {RuntimeError, "program finished with exit non-zero code " + std::to_string(proc.return_code()) + ".", 0.0, tim, memory};
1716 }
1717 return {Accepted, "program finished successfully.", 1.0, tim, memory};
1718 }
